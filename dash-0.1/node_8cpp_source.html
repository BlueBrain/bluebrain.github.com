76 template< class N, class A > inline
77 VisitorResult accept_( N* node, NodeVisitor< lunchbox::RefPtr< N >,
78  lunchbox::RefPtr< A > >& visitor )
79 {
80  switch( visitor.visitDown( node ))
81  {
82  case ACCEPT_CONTINUE: break;
83  case ACCEPT_PRUNE: return visitor.visitUp( node );
84  case ACCEPT_TERMINATE: return ACCEPT_TERMINATE;
85  }
86 
87  for( detail::ChildIterator< N > i = node->childrenBegin();
88  i < node->childrenEnd(); ++i )
89  {
90  switch( i->accept( visitor ))
91  {
92  case ACCEPT_CONTINUE: break;
93  case ACCEPT_PRUNE: return visitor.visitUp( node );
94  case ACCEPT_TERMINATE: return ACCEPT_TERMINATE;
95  }
96  }
97 
98  for( detail::AttributeIterator< N, A > i = node->attributesBegin();
99  i < node->attributesEnd(); ++i )
100  {
101  switch( visitor.visit( *i ))
102  {
103  case ACCEPT_CONTINUE: break;
104  case ACCEPT_PRUNE:
105  return visitor.visitUp( node );
106  case ACCEPT_TERMINATE: return ACCEPT_TERMINATE;
107  }
108  }
109  return visitor.visitUp( node );
110 }
111 }
