55 Context::~Context()
56 {
57  LBASSERT( _mainContext );
58  if( _currentContext == this )
59  _currentContext = 0;
60 
61  delete impl_;
62  LBASSERT( _currentContext != this );
63 
64  lunchbox::ScopedFastWrite mutex( getInitLock_( ));
65  if( _mainContext == this )
66  {
67  _mainContext = 0;
68  if( !lunchbox::exit( ))
69  LBERROR << "Collage exit failed" << std::endl;
70  }
71 }

73 Context& Context::getMain( const int argc, char** argv )
74 {
75  lunchbox::ScopedFastWrite mutex( getInitLock_( ));
76  if( !_mainContext )
77  {
78  LBASSERT( lunchbox::Version::check( ));
79  if( !lunchbox::Version::check() || !lunchbox::init( argc, argv ))
80  LBERROR << "Lunchbox initialization failed" << std::endl;
81  LBINFO << "Initializing main dash::Context v" << Version::getString()
82  << std::boolalpha << std::endl;
83  _mainContext = new Context;
84  }
85  else if( argc > 0 )
86  {
87  LBWARN << "Main context already initialized, ignoring command line "
88  << "arguments" << std::endl;
89  LBASSERT( argc == 0 );
90  }
91  return *_mainContext;
92 }

101 Context& Context::getCurrent()
102 {
103  // thread-local, therefore threadsafe
104  // Do not restructure code without making sure you don't end up with
105  // multiple pthread_getspecific calls (git blame this comment)
106  Context* context = _currentContext.get();
107  if( context )
108  return *context;
109 
110  context = &getMain();
111  _currentContext = context;
112  return *context;
113 }
