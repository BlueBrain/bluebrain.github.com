107 template<> inline std::string format( const uint8_t* data, const size_t num )
108 {
109  std::ostringstream os;
110  os << num << " bytes @ " << std::hex << (const void*)data << ": 0x";
111  os.precision( 2 );
112  for( size_t i = 0; i < num; ++i )
113  {
114  if( (i % 32) == 0 )
115  os << " " << std::endl;
116  else if( (i % 8) == 0 )
117  os << ' ';
118  os << ' ' << std::setw(2) << int( data[i] );
119  }
120  return os.str();
121 }
