44 template< class T > T* Buffer< T >::resize( const uint64_t newSize )
45 {
46  _size = newSize;
47  if( newSize <= _maxSize )
48  return _data;
49 
50  // avoid excessive reallocs
51  const uint64_t nElems = newSize + (newSize >> 3);

53  _data = static_cast< T* >( realloc( _data, nBytes ));
