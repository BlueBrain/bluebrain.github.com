301  template< class V > V _get( const std::string& k ) const
302  {
303 #if __GNUG__ && __GNUC__ < 5
304  if( !__has_trivial_copy(V) )
305 #else
306  if( !std::is_trivially_copyable< V >::value )
307 #endif
308  LBTHROW( std::runtime_error( "Can't retrieve non-POD " +
309  lunchbox::className( V( ))));
310  if( std::is_pointer< V >::value )
311  LBTHROW( std::runtime_error( "Can't retrieve pointers" ));
312 
313  const std::string& value = (*this)[ k ];
314  if( value.size() != sizeof( V ))
315  LBTHROW( std::runtime_error( std::string( "Wrong value size " ) +
316  std::to_string( value.size( )) +
317  " for type " +
318  lunchbox::className( V( ))));
319 
320  V v( *reinterpret_cast< const V* >( value.data( )));
321  if( _swap( ))
322  lunchbox::byteswap( v );
323  return v;
324  }
