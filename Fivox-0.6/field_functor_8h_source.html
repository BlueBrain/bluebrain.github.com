48 template< class TImage > inline typename FieldFunctor< TImage >::TPixel
49 FieldFunctor< TImage >::operator()( const TPoint& point, const TSpacing& ) const
50 {
51  if( !Super::_source )
52  return 0;
53 
54  const float cutOffDistance = Super::_source->getCutOffDistance();
55 
56  const size_t size = Super::_source->getNumEvents();
57  // By using the restrict keyword, we specify that the object will only
58  // be accessed by the declared pointer, which helps for the optimization
59  const float* __restrict__ posx = Super::_source->getPositionsX();
60  const float* __restrict__ posy = Super::_source->getPositionsY();
61  const float* __restrict__ posz = Super::_source->getPositionsZ();
62  const float* __restrict__ radii = Super::_source->getRadii();
63  const float* __restrict__ values = Super::_source->getValues();
64 
65  const float px( point[0] ), py( point[1] ), pz( point[2] );
66 
67  // Compute directly the inverted value to gain performance in the for loop
68  const float squaredCutoff = 1.f / ( cutOffDistance * cutOffDistance );
69  float voltage1( 0.f ), voltage2( 0.f );
70 
71  // Tell the compiler that memory accesses are aligned (done in
72  // EventSource::resize) so it is able to make optimizations
73  #pragma vector aligned
74  for( size_t i = 0; i < size; ++i )
75  {
76  const float distanceX = px - posx[i];
77  const float distanceY = py - posy[i];
78  const float distanceZ = pz - posz[i];
79 
80  const float distance2( 1.f / ( distanceX * distanceX +
81  distanceY * distanceY +
82  distanceZ * distanceZ ));
83 
84  // Comparison is inverted, as we are using the reciprocal values
85  if( distance2 < squaredCutoff )
86  continue;
87 
88  const float value( values[i] );
89 
90  // If center of the voxel within the event radius, use the
91  // voltage at the surface of the compartment (at 'radius' distance)
92  const float radius( radii[i] );
93  // Comparison is inverted, as we are using the reciprocal values
94  // (radius is already inverted from the loader)
95  if( distance2 > radius * radius )
96  voltage1 += value * radius; // mV
97  else
98  voltage2 += value * distance2; // mV
99  }
100  return voltage1 + voltage2;
101 }

103 }
