303  template <class V>
304  V _get(const std::string& k) const
305  {
306 #if __GNUG__ && __GNUC__ < 5
307  if (!__has_trivial_copy(V))
308 #else
309  if (!std::is_trivially_copyable<V>::value)
310 #endif
311  LBTHROW(std::runtime_error("Can't retrieve non-POD " +
312  lunchbox::className(V())));
313  if (std::is_pointer<V>::value)
314  LBTHROW(std::runtime_error("Can't retrieve pointers"));
315 
316  const std::string& value = (*this)[k];
317  if (value.size() != sizeof(V))
318  LBTHROW(std::runtime_error(std::string("Wrong value size ") +
319  std::to_string(value.size()) +
320  " for type " +
321  lunchbox::className(V())));
322 
323  V v(*reinterpret_cast<const V*>(value.data()));
324  if (_swap())
325  lunchbox::byteswap(v);
326  return v;
327  }

342  reinterpret_cast<const V*>(value.data() +

354 {
355  std::string value = (*this)[key];
356  V* const begin = reinterpret_cast<V*>(const_cast<char*>(value.data()));
357  V* const end = begin + value.size() / sizeof(V);
358 
359  if (_swap() && sizeof(V) != 1)
360  for (V* i = begin; i < end; ++i)
361  lunchbox::byteswap(*i);
362 
363  return std::set<V>(begin, end);
