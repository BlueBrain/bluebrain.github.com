67  template <typename T>
68  void assertValidType() const
69  {
70  static_assert(std::is_same<T, double>::value ||
71  std::is_same<T, int32_t>::value ||
72  std::is_same<T, std::string>::value ||
73  std::is_same<T, bool>::value ||
74  std::is_same<T, std::array<double, 2>>::value ||
75  std::is_same<T, std::array<int32_t, 2>>::value ||
76  std::is_same<T, std::array<double, 3>>::value ||
77  std::is_same<T, std::array<int32_t, 3>>::value ||
78  std::is_same<T, std::array<double, 4>>::value,

90  template <typename T>
91  Type getType() const
92  {
93  assertValidType<T>();
94 
95  if (std::is_same<T, double>::value)
96  return Property::Type::Double;
97  if (std::is_same<T, int32_t>::value)
98  return Property::Type::Int;
99  if (std::is_same<T, std::string>::value)
100  return Property::Type::String;
101  if (std::is_same<T, bool>::value)
102  return Property::Type::Bool;
103  if (std::is_same<T, std::array<double, 2>>::value)
104  return Property::Type::Vec2d;
105  if (std::is_same<T, std::array<int32_t, 2>>::value)
106  return Property::Type::Vec2i;
107  if (std::is_same<T, std::array<double, 3>>::value)
108  return Property::Type::Vec3d;
109  if (std::is_same<T, std::array<int32_t, 3>>::value)
110  return Property::Type::Vec3i;
111  if (std::is_same<T, std::array<double, 4>>::value)
112  return Property::Type::Vec4d;
113  throw std::runtime_error("Could not match type for '" + name + "'");
114  }

319  void setProperty(const Property& newProperty)
320  {
321  if (auto property = find(newProperty.name))
322  {
323  if (property->type != newProperty.type)
324  throw std::runtime_error(
325  "setProperty does not allow for changing the type");
