48 template <class T>
49 class Box
50 {
51 public:
52  using vec = glm::vec<3, T>;
53 
54  Box() = default;
55 
56  Box(const vec& pMin, const vec& pMax)
57  : _min(glm::min(pMin, pMax))
58  , _max(glm::max(pMin, pMax))
59  {
60  }
61  inline bool operator==(const Box<T>& other) const
62  {
63  return _min == other._min && _max == other._max;
64  }
65 
66  inline void merge(const Box<T>& aabb)
67  {
68  _min = glm::min(_min, aabb.getMin());
69  _max = glm::max(_max, aabb.getMax());
70  }
71 
72  inline void merge(const vec& point)
73  {
74  _min = glm::min(_min, point);
75  _max = glm::max(_max, point);
76  }
77 
78  inline void intersect(const Box<T>& aabb)
79  {
80  _min = glm::max(_min, aabb.getMin());
81  _max = glm::min(_max, aabb.getMax());
82  }
83 
84  inline void reset()
85  {
86  _min = vec(std::numeric_limits<T>::max());
87  _max = vec(-std::numeric_limits<T>::max());
88  }
89 
90  inline bool isEmpty() const
91  {
92  return _min.x >= _max.x || _min.y >= _max.y || _min.z >= _max.x;
93  }
94 
95  inline vec getCenter() const { return (_min + _max) * .5; }
96  inline vec getSize() const { return _max - _min; }
97  inline const vec& getMin() const { return _min; }
98  inline const vec& getMax() const { return _max; }
99 
100 #ifdef __INTEL_COMPILER // Workaround for ICC. Make members public
101 public:
102  vec _min{std::numeric_limits<T>::max()};
103  vec _max{-std::numeric_limits<T>::max()};
104 #else
105 private:
106  vec _min{std::numeric_limits<T>::max()};
107  vec _max{-std::numeric_limits<T>::max()};
108 
109  SERIALIZATION_FRIEND(Box<double>)
110  SERIALIZATION_FRIEND(Box<float>)
111 #endif
112 };
113

115 inline std::ostream& operator<<(std::ostream& os, const Box<T>& aabb)
