71 inline SDFGeometry createSDFConePill(const Vector3f& p0, const Vector3f& p1,
72  const float r0, const float r1,
73  const uint64_t data = 0)
74 {
75  SDFGeometry geom;
76  geom.userData = data;
77  geom.p0 = p0;
78  geom.p1 = p1;
79  geom.r0 = r0;
80  geom.r1 = r1;
81 
82  if (r0 < r1)
83  {
84  std::swap(geom.p0, geom.p1);
85  std::swap(geom.r0, geom.r1);
86  }
87 
88  geom.type = SDFType::ConePill;
89  return geom;
90 }

102 inline Boxd getSDFBoundingBox(const SDFGeometry& geom)
103 {
104  Boxd bounds;
105  switch (geom.type)
106  {
107  case brayns::SDFType::Sphere:
108  {
109  bounds.merge(geom.p0 - Vector3f(geom.r0));
110  bounds.merge(geom.p0 + Vector3f(geom.r0));
111  break;
112  }
113  case brayns::SDFType::Pill:
114  {
115  bounds.merge(geom.p0 - Vector3f(geom.r0));
116  bounds.merge(geom.p0 + Vector3f(geom.r0));
117  bounds.merge(geom.p1 - Vector3f(geom.r0));
118  bounds.merge(geom.p1 + Vector3f(geom.r0));
119  break;
120  }
121  case brayns::SDFType::ConePill:
122  case brayns::SDFType::ConePillSigmoid:
123  {
124  bounds.merge(geom.p0 - Vector3f(geom.r0));
125  bounds.merge(geom.p0 + Vector3f(geom.r0));
126  bounds.merge(geom.p1 - Vector3f(geom.r1));
127  bounds.merge(geom.p1 + Vector3f(geom.r1));
128  break;
129  }
130  default:
131  throw std::runtime_error("No bounds found for SDF type.");
132  }
133  return bounds;
134 }
