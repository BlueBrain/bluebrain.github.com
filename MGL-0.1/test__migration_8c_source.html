16 void vertices_into_array(dms_t m, dms_attrib_t id, dms_vertex_t* a)
17 {
18  for (dms_vertex_t temp_v = dms_roots(m);
19  temp_v != NULL; temp_v = dms_move_across(temp_v))
20  place_vertex(a,temp_v,id);
21  for (dms_vertex_t temp_v = dms_orphans(m);
22  temp_v != NULL; temp_v = dms_move_across(temp_v))
23  place_vertex(a,temp_v,id);
24 }

26 /* the graph looks like this:
27  0-->1
28  |
29  2
30  vertices have the id attribute
31  and edges have the weight attribute
32  */
33 
34 int main(int argc, char** argv)
35 {
36  debug_memory();
37  dms_start();
38  dms_t m = dms_create();
39  dms_attrib_t id = dms_add_attribute(m,DMS_INT,1);
40  dms_attrib_t weight = dms_add_attribute(m,DMS_RAT,1);
41  dms_vertex_t v[V];
42  for (long i=0; i < V; ++i)
43  v[i] = NULL;
44  dms_half_edge_t he;
45  dms_edge_t e;
46  if (dms_part() == 0)
47  {
48  // initialize the vertices and give them ID attributes
49  for (long i=0; i < V; ++i)
50  {
51  v[i] = dms_add_vertex(m);
52  dms_add_data(dms_vertex_data(v[i]),id);
53  dms_set_data(dms_vertex_data(v[i]),id,0,(union dms_value)i);
54  }
55  // add edge (0,1) with floating point data
56  e = dms_add_edge(v[0],v[1]);
57  dms_add_data(dms_edge_data(e),weight);
58  dms_set_data(dms_edge_data(e),weight,0,(union dms_value)WEIGHT);
59  // make v2 a child of v0
60  dms_make_child(v[2],v[0]);
61  dms_mark_for_migration(m);
62  // send v1 and v2 to process 1
63  dms_mark_destination(v[1],1);
64  dms_mark_destination(v[2],1);
65  assert(dms_get_destination(v[0])==0);
66  assert(dms_get_destination(v[1])==1);
67  assert(dms_get_destination(v[2])==1);
68  dms_migrate(); //migration 1
69  // assert conditions about v0
70  assert(dms_roots(m) == v[0]);
71  assert(dms_move_across(v[0]) == NULL);
72  // assert conditions on new half edge (0,1,0)
73  he = dms_half_edges(v[0],DMS_OUT);
74  assert(he != NULL);
75  assert(dms_next_half_edge(he) == NULL);
76  assert(dms_half_edge_vertex(he,DMS_IN) == v[0]);
77  assert(dms_half_edge_part(he) == 1);
78  assert(dms_has_data(dms_half_edge_data(he),weight));
79  assert(dms_get_data(dms_half_edge_data(he),weight,0).rational == WEIGHT);
80  // assert conditions on new half child 2
81  dms_half_child_t hc = dms_half_children(v[0]);
82  assert(hc != NULL);
83  assert(dms_next_half_child(hc) == NULL);
84  assert(dms_half_child_parent(hc) == v[0]);
85  assert(dms_half_child_part(hc) == 1);
86  assert(dms_has_data(dms_half_child_data(hc),id));
87  assert(dms_get_data(dms_half_child_data(hc),id,0).integer == 2);
88  // send v0 to part 1
89  dms_mark_for_migration(m);
90  dms_mark_destination(v[0],1);
91  assert(dms_get_destination(v[0])==1);
92  dms_migrate(); //migration 2
93  // assert that the graph is empty
94  assert(dms_roots(m) == NULL);
95  assert(dms_orphans(m) == NULL);
96  //receive v0 and v1 back from part 1
97  dms_mark_for_migration(m);
98  dms_migrate(); //migration 3
99  vertices_into_array(m,id,v);
100  //assert that v0 and v1 are back
101  assert(dms_roots(m) == v[1]);
102  assert(dms_move_across(v[1]) == v[0]);
103  assert(dms_move_across(v[0]) == NULL);
104  //assert that the edge (0,1) was reconstituted
105  e = dms_edges(v[0],DMS_OUT);
106  assert(e != NULL);
107  assert(e == dms_edges(v[1],DMS_IN));
108  assert(dms_next_edge(e,DMS_OUT) == NULL);
109  assert(dms_next_edge(e,DMS_IN) == NULL);
110  assert(dms_edge_vertex(e,DMS_IN) == v[0]);
111  assert(dms_edge_vertex(e,DMS_OUT) == v[1]);
112  assert(dms_get_data(dms_edge_data(e),weight,0).rational == WEIGHT);
113  }
114  else
115  {
116  dms_mark_for_migration(m);
117  dms_migrate(); //migration 1
118  // receive vertices and put them in array according to IDs
119  vertices_into_array(m,id,v);
120  // assert vertices exist
121  assert(v[1] != NULL);
122  assert(v[2] != NULL);
123  // assert conditions on new vertex v1
124  assert(dms_roots(m) == v[1]);
125  assert(dms_move_across(v[1]) == NULL);
126  // assert conditions on new half edge (0,1,1)
127  he = dms_half_edges(v[1],DMS_IN);
128  assert(he != NULL);
129  assert(dms_next_half_edge(he) == NULL);
130  assert(dms_half_edge_vertex(he,DMS_OUT) == v[1]);
131  assert(dms_half_edge_part(he) == 0);
132  assert(dms_has_data(dms_half_edge_data(he),weight));
133  assert(dms_get_data(dms_half_edge_data(he),weight,0).rational == WEIGHT);
134  // assert conditions on new cut vertex 2
135  assert(dms_orphans(m) == v[2]);
136  assert(dms_move_across(v[2]) == NULL);
137  assert(dms_is_orphan(v[2]));
138  assert(dms_orphan_part(v[2]) == 0);
139  //receive v0 from part 0
140  dms_mark_for_migration(m);
141  dms_migrate(); //migration 2
142  vertices_into_array(m,id,v);
143  // assert that the original graph is reconstituted
144  assert(dms_move_up(v[0]) == NULL);
145  assert(dms_move_up(v[1]) == NULL);
146  assert(dms_move_down(v[0]) == v[2]);
147  assert(dms_move_across(v[2]) == NULL);
148  // assert conditions on reconstituted edge
149  e = dms_edges(v[0],DMS_OUT);
150  assert(e != NULL);
151  assert(dms_next_edge(e,DMS_OUT) == NULL);
152  assert(dms_edges(v[1],DMS_IN) == e);
153  assert(dms_next_edge(e,DMS_IN) == NULL);
154  assert(dms_edge_vertex(e,DMS_IN) == v[0]);
155  assert(dms_edge_vertex(e,DMS_OUT) == v[1]);
156  assert(dms_has_data(dms_edge_data(e),weight));
157  assert(dms_get_data(dms_edge_data(e),weight,0).rational == WEIGHT);
158  //send v0 and v1 back to part 0
159  dms_mark_for_migration(m);
160  dms_mark_destination(v[0],0);
161  dms_mark_destination(v[1],0);
162  assert(dms_get_destination(v[0])==0);
163  assert(dms_get_destination(v[1])==0);
164  assert(dms_get_destination(v[2])==1);
165  dms_migrate(); //migration 3
166  vertices_into_array(m,id,v);
167  //assert that only orphan v2 is left
168  assert(dms_move_across(v[2]) == NULL);
169  assert(dms_is_orphan(v[2]));
170  assert(dms_orphans(m) == v[2]);
171  assert(dms_roots(m) == NULL);
172  }
173  dms_delete(m);
174  dms_stop();
175  return 0;
176 }
