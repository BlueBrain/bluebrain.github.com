34 void ms_delete(struct multiscale* m)
35 {
36  struct vertex* v;
37  while ((v = vertices(m)))
38  delete_vertex(v);
39  struct attribute* a;
40  while ((a = ms_begin_attributes(m)))
41  {
42  remove_attribute(a);

44  }
45  reallocate(m,0);
46 }

164 union attribute_value api_to_internal(union ms_value in, enum attribute_type type)
165 {
166  union attribute_value out;
167  if (type == boolean_attribute)
168  out.boolean = in.boolean;
169  else if (type == integer_attribute)
170  out.integer = in.integer;
171  else if (type == rational_attribute)
172  out.rational = in.rational;
173  else if (type == text_attribute)
174  out.text = in.text;
