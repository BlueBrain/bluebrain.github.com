94 union attribute_value get_attribute(struct attribute* a, struct attributed_data* d, natural_t element)
95 {
96  union attribute_value val;
97  union word* entry = d->entries[a->index];
98  if (a->type == boolean_attribute)
99  {
100  natural_t bit = element % WORD_BITS;
101  natural_t index = element / WORD_BITS;
102  val.boolean = GET_BIT(entry[index].natural,bit);
103  }
104  else if (a->type == integer_attribute)
105  {
106  val.integer = entry[element].integer;
107  }
108  else if (a->type == rational_attribute)
109  {
110  val.rational = entry[element].rational;
111  }
112  else if (a->type == text_attribute)
113  {
114  natural_t byte = element%WORD_SIZE;
115  natural_t index = element/WORD_SIZE;
116  val.text = entry[index].text[byte];
117  }
118  return val;
119 }

149 void copy_attribute_data(struct attributes* as, struct attributed_data* from, struct attributed_data* to)
150 {
151  destroy_attributed_data(to);
152  to->size = from->size;
153  to->entries = reallocate(NULL,WORD_SIZE*(to->size));
154  natural_t words;
155  for (struct attribute* a = get_attributes(as);
156  a != NULL;
157  a = next_attribute(a))
158  {
159  if (a->index >= to->size)
160  continue;
161  if (from->entries[a->index] == NULL)
162  to->entries[a->index] = NULL;
163  else
164  {
165  words = attribute_words(a);
166  to->entries[a->index] = reallocate(NULL,WORD_SIZE*words);
167  memcpy(to->entries[a->index],from->entries[a->index],words*WORD_SIZE);
168  }
169  }
170 }
