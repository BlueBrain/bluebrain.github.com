30 struct avl_node* avl_rotate(struct avl_node* from, enum bi_move side)
31 {
32  struct avl_node* second_node;
33  second_node = avl_move(from,side);
34  bool zero_case = (second_node->balance == 0);
35  if ( ! zero_case)
36  {
37  enum bi_move second_side = INT_TO_SIGN(second_node->balance);
38  if (second_side != side)
39  rotate_bi_node(&(second_node->bi_node),1-second_side);
40  }
41  struct avl_node* result;
42  result = (struct avl_node*) rotate_bi_node(&(from->bi_node),1-side);
43  if (zero_case)
44  {
45  from->balance = SIGN_TO_INT(side);
46  second_node->balance = -(from->balance);
47  }
48  else
49  {
50  from->balance = 0;
51  second_node->balance = 0;
52  }
53  return result;
54 }

56 struct avl_node* balance_avl_nodes(struct avl_node* node, short change)
57 {
58  struct avl_node* parent = node;
59  short side_factor;
60  while (change != 0)
61  {
62  node = parent;
63  parent = avl_move(node,bi_up);
64  if (parent == NULL)
65  return node;
66  side_factor = SIGN_TO_INT(bi_child_side(&(parent->bi_node),&(node->bi_node)));
67  parent->balance += change*side_factor;
68  if (abs(parent->balance) == 2)
69  parent = avl_rotate(parent,INT_TO_SIGN(parent->balance));
70  if (abs(parent->balance) == INT_TO_SIGN(-change))
71  return parent;
72  }
73  return parent;
74 }
