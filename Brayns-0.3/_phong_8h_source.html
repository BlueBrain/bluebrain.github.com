97 static __device__ float3 refractedVector(const float3 direction,
98  const float3 normal, const float n1,
99  const float n2)
100 {
101  if (n2 < epsilon || n1 == n2)
102  return direction;
103  const float eta = n1 / n2;
104  const float cos1 = -optix::dot(direction, normal);
105  const float cos2 = 1.f - eta * eta * (1.f - cos1 * cos1);
106  if (cos2 > 0.f)
107  return eta * direction + (eta * cos1 - sqrt(cos2)) * normal;
108  return direction;
109 }

111 static __device__ float4 getVolumeContribution(optix::Ray volumeRay,
112  int iteration)
113 {
114  float pathAlpha = 0.f;
115  float3 pathColor = make_float3(0.f);
116  float3 lightContribution = make_float3(1.f);
117 
118  optix::size_t2 screen = output_buffer.size();
119  unsigned int seed =
120  tea<16>(screen.x * launch_index.y + launch_index.x, frame);
121  const float delta = (float)(seed % (int)(volumeEpsilon + 1));
122 
123  float t = volumeEpsilon - delta;
124  const float tMax = t_hit - volumeEpsilon + volumeDiag;
125 
126  while (t < tMax && pathAlpha < 1.f)
127  {
128  const float x = rnd(seed) * volumeEpsilon;
129  float3 point = ((volumeRay.origin + volumeRay.direction * (t + x)) -
130  volumeOffset) /
131  volumeElementSpacing;
132 
133  if (point.x > 0.f && point.x < volumeDimensions.x && point.y > 0.f &&
134  point.y < volumeDimensions.y && point.z > 0.f &&
135  point.z < volumeDimensions.z)
136  {
137  ulong index = (ulong)((ulong)floor(point.x) +
138  (ulong)floor(point.y) * volumeDimensions.x +
139  (ulong)floor(point.z) * volumeDimensions.x *
140  volumeDimensions.y);
141 
142  const unsigned char voxelValue = volumeData[index];
143 
144  // Determine light contribution
145  if (shadows != 0.f && iteration > 0)
146  {
147  unsigned int num_lights = lights.size();
148  for (int i = 0; i < num_lights; ++i)
149  {
150  optix::Ray lightRay = volumeRay;
151  lightRay.origin = point;
152 
153  BasicLight light = lights[i];
154  if (light.type == 0)
155  {
156  // Point light
157  float3 pos = light.pos;
158  if (soft_shadows > 0.f)
159  // Soft shadows
160  pos += soft_shadows * make_float3(rnd(seed) - 0.5f,
161  rnd(seed) - 0.5f,
162  rnd(seed) - 0.5f);
163  lightRay.direction = optix::normalize(pos - point);
164  }
165  else
166  {
167  // Directional light
168  float3 L = -light.dir;
169  if (soft_shadows > 0.f)
170  // Soft shadows
171  L += soft_shadows * make_float3(rnd(seed) - 0.5f,
172  rnd(seed) - 0.5f,
173  rnd(seed) - 0.5f);
174  lightRay.direction = optix::normalize(L);
175  }
176  const float4 voxelColor =
177  getVolumeContribution(lightRay, iteration - 1);
178  lightContribution =
179  lightContribution - make_float3(voxelColor);
180  pathAlpha += voxelColor.w;
181  }
182  }
183  lightContribution = make_float3(max(0.f, lightContribution.z),
184  max(0.f, lightContribution.y),
185  max(0.f, lightContribution.x));
186  pathAlpha = min(1.f, pathAlpha);
187 
188  if (colorMapSize != 0)
189  {
190  const float normalizedValue = colorMapSize *
191  (voxelValue - colorMapMinValue) /
192  colorMapRange;
193  const float4 colorMapColor = colorMap[normalizedValue];
194  const float3 emissionIntensity =
195  emissionIntensityMap[normalizedValue];
196 
197  const float3 voxelColor =
198  (make_float3(emissionIntensity.z, emissionIntensity.y,
199  emissionIntensity.x) +
200  lightContribution) *
201  make_float3(colorMapColor.z, colorMapColor.y,
202  colorMapColor.x);
203 
204  const float alphaMagic =
205  magicExponent * volumeEpsilon / volumeDiag;
206  const float alpha =
207  1.f - pow(1.f - min(colorMapColor.w,
208  1.f - 1.f / (float)colorMapSize),
209  alphaMagic);
210  pathColor =
211  pathColor + (voxelColor * alpha * (1.f - pathAlpha));
212  pathAlpha = pathAlpha + (alpha * (1.f - pathAlpha));
213  }
214  else
215  {
216  const float3 voxelColor = {point.x, point.y, point.z};
217  const float alpha = 1.f;
218  pathColor =
219  pathColor + (voxelColor * alpha * (1.f - pathAlpha));
220  pathAlpha = pathAlpha + (alpha * (1.f - pathAlpha));
221  }
222  }
223  t += volumeEpsilon;
224  }
225  if (shadows != 0.f && iteration > 0)
226  pathColor = pathColor * defaultVolumeAlpha;
227  return make_float4(min(1.f, pathColor.x), min(1.f, pathColor.y),
228  min(1.f, pathColor.z), min(1.f, pathAlpha));
229 }

238 static __device__ void phongShade(float3 p_Kd, float3 p_Ka, float3 p_Ks,
239  float3 p_Kr, float3 p_Ko,
240  float p_refractionIndex, float p_phong_exp,
241  float p_glossiness, float3 p_normal,
242  float p_ray_tmax)
243 {
244  // Randomness
245  optix::size_t2 screen = output_buffer.size();
246  unsigned int seed =
247  tea<16>(screen.x * launch_index.y + launch_index.x, frame);
248 
249  // Glossiness
250  if (p_glossiness > 0.f)
251  p_normal = optix::normalize(
252  p_normal +
253  p_glossiness * make_float3(rnd(seed) - 0.5f, rnd(seed) - 0.5f,
254  rnd(seed) - 0.5f));
255  // Ambient contribution
256  float3 result = {0.f, 0.f, 0.f};
257 
258  // Volume
259  float4 volumeValue = {0.f, 0.f, 0.f, 0.f};
260  if (volumeDiag != 0.f)
261  {
262  optix::Ray volumeRay = ray;
263  volumeValue = getVolumeContribution(volumeRay, 1);
264  result.x += volumeValue.x;
265  result.y += volumeValue.y;
266  result.z += volumeValue.z;
267  result += (1.f - volumeValue.w) * bg_color;
268  }
269 
270  // Surface
271  float3 hit_point = ray.origin + t_hit * ray.direction;
272 
273  float3 light_attenuation = make_float3(1.f - volumeValue.w);
274 
275  if (fmaxf(p_Ko) < epsilon)
276  result += p_Ko * p_Kd;
277  else if (electron_shading_enabled == 1)
278  result +=
279  p_Kd * (1.f - abs(optix::dot(optix::normalize(hit_point - eye),
280  p_normal)));
281  else if (shading_enabled == 0)
282  result += p_Kd;
283  else
284  {
285  // compute direct lighting
286  unsigned int num_lights = lights.size();
287  for (int i = 0; i < num_lights; ++i)
288  {
289  BasicLight light = lights[i];
290 
291  float3 L;
292  float Ldist = p_ray_tmax;
293  if (light.type == 0)
294  {
295  // Point light
296  float3 pos = light.pos;
297  if (shadows > 0.f && soft_shadows > 0.f)
298  // Soft shadows
299  pos += soft_shadows * make_float3(rnd(seed) - 0.5f,
300  rnd(seed) - 0.5f,
301  rnd(seed) - 0.5f);
302  L = optix::normalize(pos - hit_point);
303  Ldist = optix::length(pos - hit_point);
304  }
305  else
306  {
307  // Directional light
308  L = -light.dir;
309  if (shadows > 0.f && soft_shadows > 0.f)
310  // Soft shadows
311  L += soft_shadows * make_float3(rnd(seed) - 0.5f,
312  rnd(seed) - 0.5f,
313  rnd(seed) - 0.5f);
314  L = optix::normalize(L);
315  }
316  float nDl = optix::dot(p_normal, L);
317 
318  // Shadows
319  if (shadows > 0.f)
320  {
321  light_attenuation = make_float3(static_cast<float>(nDl > 0.0f));
322  if (nDl > 0.f && light.casts_shadow)
323  {
324  PerRayData_shadow shadow_prd;
325  shadow_prd.attenuation = make_float3(1.f);
326  hit_point =
327  ray.origin + t_hit * ray.direction + scene_epsilon * L;
328  optix::Ray shadow_ray =
329  optix::make_Ray(hit_point, L, shadow_ray_type,
330  scene_epsilon, Ldist);
331  rtTrace(top_shadower, shadow_ray, shadow_prd);
332  // light_attenuation is zero if completely shadowed
333  light_attenuation = shadow_prd.attenuation;
334  }
335  }
336 
337  if (fmaxf(light_attenuation) < epsilon)
338  light_attenuation = make_float3(1.f - shadows);
339 
340  // If not completely shadowed, light the hit point
341  if (fmaxf(light_attenuation) > 0.f)
342  {
343  float3 Lc = light.color * light_attenuation;
344 
345  result += p_Kd * nDl * Lc;
346 
347  float3 H = optix::normalize(L - ray.direction);
348  float nDh = optix::dot(p_normal, H);
349  if (nDh > 0.f)
350  {
351  // Specular
352  float power = pow(nDh, p_phong_exp);
353  result += p_Ks * power * Lc;
354  }
355  }
356  }
357  }
358 
359  // Reflection
360  if (fmaxf(p_Kr) > epsilon)
361  {
362  PerRayData_radiance new_prd;
363  new_prd.importance = prd.importance * optix::luminance(p_Kr);
364  new_prd.depth = prd.depth + 1;
365 
366  if (new_prd.importance > epsilon && new_prd.depth <= max_depth)
367  {
368  float3 R = optix::reflect(ray.direction, p_normal);
369  hit_point = ray.origin + t_hit * ray.direction + scene_epsilon * R;
370  optix::Ray refl_ray =
371  optix::make_Ray(hit_point, R, radiance_ray_type, scene_epsilon,
372  p_ray_tmax);
373  rtTrace(top_object, refl_ray, new_prd);
374  result += light_attenuation * p_Kr * new_prd.result;
375  }
376  }
377 
378  // Refraction
379  if (fmaxf(p_Ko) < 1.f)
380  {
381  PerRayData_radiance new_prd;
382  new_prd.importance =
383  max(0.f, prd.importance - optix::luminance(1.f - p_Ko));
384  new_prd.depth = prd.depth + 1;
385 
386  if (new_prd.importance > epsilon && new_prd.depth <= max_depth)
387  {
388  float3 R = refractedVector(ray.direction, p_normal,
389  p_refractionIndex, 1.f);
390  hit_point = ray.origin + t_hit * ray.direction + scene_epsilon * R;
391  optix::Ray refr_ray =
392  optix::make_Ray(hit_point, R, radiance_ray_type, scene_epsilon,
393  p_ray_tmax);
394  rtTrace(top_object, refr_ray, new_prd);
395  result += light_attenuation * (1.f - p_Ko) * new_prd.result;
396  }
397  }
398 
399  // Only opaque surfaces are affected by Global Illumination
400  if (fmaxf(p_Ko) == 1.f)
401  {
402  // Ambient occlusion
403  if (ambient_occlusion_strength > 0.f)
404  {
405  PerRayData_shadow aa_prd;
406  aa_prd.attenuation = make_float3(1.f);
407 
408  float3 aa_normal =
409  optix::normalize(make_float3(rnd(seed) - 0.5f, rnd(seed) - 0.5f,
410  rnd(seed) - 0.5f));
411  if (optix::dot(aa_normal, p_normal) < 0.f)
412  aa_normal = -aa_normal;
413  hit_point =
414  ray.origin + t_hit * ray.direction + scene_epsilon * aa_normal;
415  optix::Ray aa_ray =
416  optix::make_Ray(hit_point, aa_normal, shadow_ray_type,
417  scene_epsilon, p_ray_tmax);
418  rtTrace(top_shadower, aa_ray, aa_prd);
419  result *=
420  1.f - ambient_occlusion_strength * (1.f - aa_prd.attenuation);
421  }
422 
423  // Radiosity
424  if (ambient_occlusion_strength > 0.f && prd.depth == 0)
425  {
426  PerRayData_radiance new_prd;
427  new_prd.importance = prd.importance;
428  new_prd.depth = prd.depth + 1;
429  if (new_prd.importance > epsilon && new_prd.depth <= max_depth)
430  {
431  float3 ra_normal = optix::normalize(
432  make_float3(rnd(seed) - 0.5f, rnd(seed) - 0.5f,
433  rnd(seed) - 0.5f));
434  if (optix::dot(ra_normal, p_normal) < 0.f)
435  ra_normal = -ra_normal;
436  hit_point = ray.origin + t_hit * ray.direction +
437  scene_epsilon * ra_normal;
438  optix::Ray ra_ray =
439  optix::make_Ray(hit_point, ra_normal, radiance_ray_type,
440  scene_epsilon, p_ray_tmax);
441  rtTrace(top_shadower, ra_ray, new_prd);
442  result += light_attenuation * ambient_occlusion_strength *
443  new_prd.result;
444  }
445  }
446  }
447  prd.result = result;
448 }
