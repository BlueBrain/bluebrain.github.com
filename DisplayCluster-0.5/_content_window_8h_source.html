239  template< class Archive >
240  void serialize_members_xml( Archive & ar, const unsigned int version )
241  {
242  ar & boost::serialization::make_nvp( "content", content_ );
243  if( version < 1 )
244  {
245  int contentWidth = 0, contentHeight = 0;
246  ar & boost::serialization::make_nvp( "contentWidth", contentWidth );
247  ar & boost::serialization::make_nvp( "contentHeight", contentHeight );
248  }
249  ar & boost::serialization::make_nvp( "coordinates", coordinates_ );
250  ar & boost::serialization::make_nvp( "coordinatesBackup", coordinatesBackup_ );
251  QPointF zoomCenter = zoomRect_.center();
252  qreal zoom = 1.0/zoomRect_.width();
253  ar & boost::serialization::make_nvp( "centerX", zoomCenter.rx( ));
254  ar & boost::serialization::make_nvp( "centerY", zoomCenter.ry( ));
255  ar & boost::serialization::make_nvp( "zoom", zoom );
256  QRectF zoomRect;
257  zoomRect.setSize( QSizeF( 1.0/zoom, 1.0/zoom ));
258  zoomRect.moveCenter( zoomCenter );
259  setZoomRect( zoomRect );
260  if( version < 1 )
261  {
262  int controlState = 0;
263  ar & boost::serialization::make_nvp( "controlState", controlState );
264  }
265  ar & boost::serialization::make_nvp( "windowState", windowState_ );
266  }
