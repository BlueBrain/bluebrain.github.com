207  template< typename C > inline void
208  DataOStream::serializeChildren( const std::vector<C*>& children )
209  {
210  const uint64_t nElems = children.size();
211  (*this) << nElems;
212 
213  for( typename std::vector< C* >::const_iterator i = children.begin();
214  i != children.end(); ++i )
215  {
216  C* child = *i;
217  (*this) << ObjectVersion( child );
218  LBASSERTINFO( !child || child->isAttached(),
219  "Found unmapped object during serialization" );
220  }
221  }
