128 void cut_pin(struct pin* p, struct needle* n, struct pin_head* h)
129 {
130  struct cut_cushion* c = pin_cut_cushion(p);
131  pull_pin(p);
132  if (c != NULL)
133  push_needle(n,c);
134  cut(&(n->cut),&(h->cut));
135 }
