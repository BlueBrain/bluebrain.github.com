206  template< typename O, typename C > inline void
207  DataIStream::deserializeChildren( O* object, const std::vector< C* >& old_,
208  std::vector< C* >& result )
209  {
210  ObjectVersions versions;
211  (*this) >> versions;
212  std::vector< C* > old = old_;
213 
214  // rebuild vector from serialized list
215  result.clear();
216  for( ObjectVersions::const_iterator i = versions.begin();
217  i != versions.end(); ++i )
218  {
219  const ObjectVersion& version = *i;
220 
221  if( version.identifier == UUID::ZERO )
222  {
223  result.push_back( 0 );
224  continue;
225  }
226 
227  typename std::vector< C* >::iterator j =
228  stde::find_if( old, ObjectFinder( version.identifier ));
229 
230  if( j == old.end( )) // previously unknown child
231  {
232  C* child = 0;
233  object->create( &child );
234  LocalNodePtr localNode = object->getLocalNode();
235  LBASSERT( child );
236  LBASSERT( !object->isMaster( ));
237 
238  LBCHECK( localNode->mapObject( child, version ));
239  result.push_back( child );
240  }
241  else
242  {
243  C* child = *j;
244  old.erase( j );
245  if( object->isMaster( ))
246  child->sync( VERSION_HEAD );
247  else
248  child->sync( version.version );
249 
250  result.push_back( child );
251  }
252  }
253 
254  while( !old.empty( )) // removed children
255  {
256  C* child = old.back();
257  old.pop_back();
258  if( !child )
259  continue;
260 
261  if( child->isAttached() && !child->isMaster( ))
262  {
263  LocalNodePtr localNode = object->getLocalNode();
264  localNode->unmapObject( child );
265  }
266  object->release( child );
267  }
268  }
