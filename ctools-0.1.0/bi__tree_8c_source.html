44 struct bi_node* insert_bi_node(struct bi_tree* t, struct bi_node* n, bi_compare_t compare)
45 {
46  if (t->root == NULL)
47  {
48  t->root = n;
49  return n;
50  }
51  struct bi_node* it;
52  struct bi_node* parent;
53  it = t->root;
54  enum bi_comparison comparison;
55  do
56  {
57  parent = it;
58  comparison = compare(n,it);
59  if (comparison == bi_equal)
60  return it;
61  it = it->next[comparison-1];
62  } while (it != NULL);
63  add_bi_child(parent,n,comparison-1);
64  return n;
65 }

113 struct bi_node* rotate_bi_node(struct bi_node* root, enum bi_move dir)
114 {
115  struct bi_node temp;
116  struct bi_node temp_low;
117  struct bi_node* pivot;
118  struct bi_node* low;
119  create_bi_node(&temp);
120  pivot = root->next[1-dir];
121  low = pivot->next[dir];
122  if (low == NULL)
123  {
124  create_bi_node(&temp_low);
125  low = &temp_low;
126  add_bi_child(pivot,low,dir);
127  }
128  change_bi_child(root,&temp);
129  change_bi_child(low,root);
130  change_bi_child(pivot,low);
131  change_bi_child(&temp,pivot);
132  if (low == &temp_low)
133  remove_bi_child(root,low,1-dir);
134  return pivot;
135 }
