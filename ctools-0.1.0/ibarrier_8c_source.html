27 void ibarrier_begin()
28 {
29  create_message(&(global_ibarrier.message));
30  global_ibarrier.data = 'b';
31  global_ibarrier.message.buffer.data = &(global_ibarrier.data);
32  global_ibarrier.message.buffer.size = sizeof(global_ibarrier.data);
33  global_ibarrier.power = 1;
34  global_ibarrier.self = parallel_self();
35  global_ibarrier.max = parallel_count()-1;
36  global_ibarrier.state = ibarrier_in;
37  ibarrier_start_step();
38 }

50 bool end_of_ibarrier_in()
51 {
52  if (global_ibarrier.self == 0) //zero special case: stop at highest bit of max value
53  return (global_ibarrier.power << 1) > global_ibarrier.max;
54  return ibarrier_bit() == 1;
55 }

57 void ibarrier_transition()
58 {
59  if (global_ibarrier.state == ibarrier_in)
60  {
61  if (end_of_ibarrier_in())
62  {
63  global_ibarrier.state = ibarrier_out;
64  }
65  else
66  {
67  global_ibarrier.power <<= 1;
68  }
69  }
70  else if (global_ibarrier.state == ibarrier_out)
71  {
72  global_ibarrier.power >>= 1;
73  if (global_ibarrier.power == 0)
74  {
75  global_ibarrier.state = ibarrier_idle;
76  }
77  }
78 }

92 void ibarrier_send_or_receive(enum ibarrier_op op)
93 {
94  if (op == ibarrier_send)

96  parallel_send(&(global_ibarrier.message),ibarrier_channel);

109 void ibarrier_start_step()
110 {
111  global_ibarrier.message.other = POWER_FLIP(global_ibarrier.self,global_ibarrier.power);
112  if (ibarrier_fake_message())
113  return;
114  ibarrier_send_or_receive(op_table[global_ibarrier.state][ibarrier_bit()]);
115 }

117 bool ibarrier_done()
118 {
119  if (global_ibarrier.state == ibarrier_idle)
120  return true;
121  if ( ! ibarrier_step_done())
122  return false;
123  ibarrier_transition();
124  if (global_ibarrier.state == ibarrier_idle)
125  return true;
126  ibarrier_start_step();
127  return false;
128 }
