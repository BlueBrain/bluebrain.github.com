41 bool bsp_receive(struct message* m)
42 {
43  if (the_receive_state == bsp_probing)
44  {
45  if ( ! parallel_try_receive(m,user_channel))
46  return false;
47  the_receive_state = bsp_reading;

52  the_receive_state = bsp_probing;
