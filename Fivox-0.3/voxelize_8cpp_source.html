71 template< typename T >
72 void _sample( ImageSourcePtr source, const vmml::Vector2ui& frameRange,
73  const bool vsdProjection, const float volumeResolution,
74  const std::string& outputFile )
75 {
76  VolumePtr input = source->GetOutput();
77  VolumeWriter< T > writer( input );
78 
79  const size_t numDigits = std::to_string( frameRange.y( )).length( );
80  for( uint32_t i = frameRange.x(); i < frameRange.y(); ++i )
81  {
82  std::string filename;
83  if( frameRange.y() - frameRange.x() > 1 )
84  {
85  std::ostringstream fileStream;
86  fileStream << outputFile << std::setfill('0')
87  << std::setw( numDigits ) << i;
88  filename = fileStream.str();
89  }
90  else
91  filename = outputFile;
92 
93  source->getFunctor()->getSource()->load( i );
94 
95  const std::string& volumeName = filename + ".mhd";
96  writer->SetFileName( volumeName );
97  source->Modified();
98  writer->Update(); // Run pipeline to write volume
99  LBINFO << "Volume written as " << volumeName << std::endl;
100 
101  if( !vsdProjection )
102  continue;
103 
104  // The projection filter computes the output using the real value of
105  // the data, i.e. not limited by the precision of the final image
106  typedef fivox::BeerLambertProjectionImageFilter
107  < Volume, FloatImageType > FilterType;
108  FilterType::Pointer projection = FilterType::New();
109  projection->SetInput( input );
110  projection->SetProjectionDimension( 1 ); // projection along Y-axis
111  projection->SetPixelSize( 1.0 / volumeResolution );
112  projection->SetSigma( sigmaVSDProjection );
113 
114  // Write output image
115  typedef itk::ImageFileWriter< FloatImageType > ImageWriter;
116  ImageWriter::Pointer imageWriter = ImageWriter::New();
117  imageWriter->SetInput( projection->GetOutput( ));
118 
119  const std::string& imageFile = filename + ".vtk";
120  imageWriter->SetFileName( imageFile );
121  imageWriter->Update();
122  LBINFO << "VSD projection written as " << imageFile << std::endl;
123  }
124 }
125 }
