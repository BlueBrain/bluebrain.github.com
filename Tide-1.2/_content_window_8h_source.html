308  template< class Archive >
309  void serialize_members_xml( Archive & ar, const unsigned int version )
310  {
311  ar & boost::serialization::make_nvp( "content", _content );
312  if( version < 1 )
313  {
314  int contentWidth = 0, contentHeight = 0;
315  ar & boost::serialization::make_nvp( "contentWidth", contentWidth );
316  ar & boost::serialization::make_nvp( "contentHeight", contentHeight );
317  }
318  ar & boost::serialization::make_nvp( "coordinates", _coordinates );
319  if( version < 3 )
320  {
321  QRectF backup;
322  ar & boost::serialization::make_nvp( "coordinatesBackup", backup );
323  }
324  QPointF zoomCenter = _content->getZoomRect().center();
325  qreal zoom = 1.0 / _content->getZoomRect().width();
326  ar & boost::serialization::make_nvp( "centerX", zoomCenter.rx( ));
327  ar & boost::serialization::make_nvp( "centerY", zoomCenter.ry( ));
328  ar & boost::serialization::make_nvp( "zoom", zoom );
329  if( version >= 3 )
330  {
331  bool focused = isFocused();
332  ar & boost::serialization::make_nvp( "focused", focused );
333  if( focused )
334  setMode( WindowMode::FOCUSED );
335  }
336  QRectF zoomRect;
337  zoomRect.setSize( QSizeF( 1.0/zoom, 1.0/zoom ));
338  zoomRect.moveCenter( zoomCenter );
339  _content->setZoomRect( zoomRect );
340  if( version < 1 )
341  {
342  int controlState = 0;
343  ar & boost::serialization::make_nvp( "controlState", controlState );
344  }
345  ar & boost::serialization::make_nvp( "windowState", _windowState );
346  }
