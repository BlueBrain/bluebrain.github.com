78  template <typename Params, typename RetVal>
79  ClientRequest<RetVal> request(const std::string& method,
80  const Params& params)
81  {
82  auto promise = std::make_shared<std::promise<RetVal>>();
83  auto callback = [promise](Response response) {
84  if (response.isError())
85  promise->set_exception(
86  std::make_exception_ptr(response_error(response.error)));
87  else
88  {
89  RetVal value;
90  if (!from_json(value, response.result))
91  promise->set_exception(jsonConversionFailed());
92  else
93  promise->set_value(std::move(value));
94  }
95  };
96  return {request(method, to_json(params), callback),
97  promise->get_future(),
98  [this](std::string method_, std::string params_) {
99  notify(method_, params_);
100  }};
101  }
