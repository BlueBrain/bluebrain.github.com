147  template <typename Params, typename RetVal>
148  void bind(const std::string& method, std::function<RetVal(Params)> action)
149  {
150  bind(method, [action](const Request& request) {
151  Params params;
152  if (!from_json(params, request.message))
153  return Response::invalidParams();
154  try
155  {
156  const auto& ret = action(std::move(params));
157  return Response{to_json(ret)};
158  }
159  catch (const response_error& e)
160  {
161  return Response{Response::Error{e.what(), e.code}};
162  }
163  });
164  }
