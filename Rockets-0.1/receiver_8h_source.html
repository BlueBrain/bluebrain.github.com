134  template <typename Params, typename RetVal>
135  void bind(const std::string& method, std::function<RetVal(Params)> action)
136  {
137  bind(method, [action](const Request& request) {
138  Params params;
139  if (!from_json(params, request.message))
140  return Response::invalidParams();
141  try
142  {
143  const auto& ret = action(std::move(params));
144  return Response{to_json(ret)};
145  }
146  catch (const response_error& e)
147  {
148  return Response{Response::Error{e.what(), e.code}};
149  }
150  });
151  }
