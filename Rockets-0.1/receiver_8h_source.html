144  template <typename Params, typename RetVal>
145  void bind(const std::string& method, std::function<RetVal(Params)> action)
146  {
147  bind(method, [action](const Request& request) {
148  Params params;
149  if (!from_json(params, request.message))
150  return Response::invalidParams();
151  try
152  {
153  const auto& ret = action(std::move(params));
154  return Response{to_json(ret)};
155  }
156  catch (const response_error& e)
157  {
158  return Response{Response::Error{e.what(), e.code}};
159  }
160  });
161  }
