87  template <typename T>
88  Type getType() const
89  {
90  assertValidType<T>();
91 
92  if (std::is_same<T, double>::value)
93  return Property::Type::Double;
94  if (std::is_same<T, int32_t>::value)
95  return Property::Type::Int;
96  if (std::is_same<T, std::string>::value)
97  return Property::Type::String;
98  if (std::is_same<T, bool>::value)
99  return Property::Type::Bool;
100  if (std::is_same<T, std::array<double, 2>>::value)
101  return Property::Type::Vec2d;
102  if (std::is_same<T, std::array<int32_t, 2>>::value)
103  return Property::Type::Vec2i;
104  if (std::is_same<T, std::array<double, 3>>::value)
105  return Property::Type::Vec3d;
106  if (std::is_same<T, std::array<int32_t, 3>>::value)
107  return Property::Type::Vec3i;
108  if (std::is_same<T, std::array<double, 4>>::value)
109  return Property::Type::Vec4d;
110  throw std::runtime_error("Could not match type for '" + name + "'");
111  }
