155 {
156 template <typename T>
157 inline std::ostream& operator<<(std::ostream& os,
158  const boost::multi_array<T, 2>& data)
159 {
160  for (size_t i = 0; i < data.shape()[0]; ++i)
161  {
162  for (size_t j = 0; j < data.shape()[1]; ++j)
163  os << data[i][j] << " ";
164  os << std::endl;
165  }
166  return os;
167 }
168 }
