31 void _serializeArray(uint8_t*& dst, const std::vector<T>& src)
32 {
33  const uint64_t arraySize = src.size();
34  *reinterpret_cast<uint64_t*>(dst) = arraySize;
35  dst += sizeof(uint64_t);
36 
37  if (arraySize > 0)
38  {
39  memcpy(dst, src.data(), sizeof(T) * src.size());
40  dst += sizeof(T) * src.size();
41  }
42 }

44 template <typename T>
45 bool _deserializeArray(std::vector<T>& dst, const uint8_t*& src,
46  const uint8_t* end)
47 {
48  if (src + sizeof(uint64_t) > end)
49  return false;
50  const uint64_t arraySize = *reinterpret_cast<const uint64_t*>(src);
51  src += sizeof(uint64_t);
52 
53  if (arraySize == 0)
54  {
55  dst.clear();
56  return true;
57  }
58 
59  if (src + sizeof(T) * arraySize > end)
60  return false;
61  const T* srcPtr = reinterpret_cast<const T*>(src);
62  dst.assign(srcPtr, srcPtr + arraySize);
63  src += sizeof(T) * arraySize;
64  return true;
65 }
66 }

94 {
95  const uint8_t* ptr = reinterpret_cast<const uint8_t*>(data);
96  const uint8_t* const end = ptr + size;
97  if (size < sizeof(MorphologyVersion) + sizeof(CellFamily))
98  return false;

100  _data.version = *reinterpret_cast<const MorphologyVersion*>(ptr);
101  ptr += sizeof(MorphologyVersion);
102 
103  _data.family = *reinterpret_cast<const CellFamily*>(ptr);
104  ptr += sizeof(CellFamily);
105 
106  if (_deserializeArray(_points, ptr, end) &&
107  _deserializeArray(_sections, ptr, end) &&
108  _deserializeArray(_sectionTypes, ptr, end) &&
109  _deserializeArray(_perimeters, ptr, end))
110  {
111  return true;
112  }
113

115  _sections.clear();
116  _sectionTypes.clear();
117  _perimeters.clear();
