00031 {
00032     struct avl_node* second_node;
00033     second_node = avl_move(from,side);
00034     bool zero_case = (second_node->balance == 0);
00035     if ( ! zero_case)
00036     {
00037         enum bi_move second_side = INT_TO_SIGN(second_node->balance);
00038         if (second_side != side)
00039             rotate_bi_node(&(second_node->bi_node),1-second_side);
00040     }
00041     struct avl_node* result;
00042     result = (struct avl_node*) rotate_bi_node(&(from->bi_node),1-side);
00043     if (zero_case)
00044     {
00045         from->balance = SIGN_TO_INT(side);
00046         second_node->balance = -(from->balance);
00047     }
00048     else
00049     {
00050         from->balance = 0;
00051         second_node->balance = 0;
00052     }
00053     return result;

00056 struct avl_node* balance_avl_nodes(struct avl_node* node, short change)
00057 {
00058     struct avl_node* parent = node;
00059     short side_factor;
00060     while (change != 0)
00061     {
00062         node = parent;
00063         parent = avl_move(node,bi_up);
00064         if (parent == NULL)
00065             return node;
00066         side_factor = SIGN_TO_INT(bi_child_side(&(parent->bi_node),&(node->bi_node)));
00067         parent->balance += change*side_factor;
00068         if (abs(parent->balance) == 2)
00069             parent = avl_rotate(parent,INT_TO_SIGN(parent->balance));
00070         if (abs(parent->balance) == INT_TO_SIGN(-change))
00071             return parent;
00072     }
00073     return parent;
00074 }
