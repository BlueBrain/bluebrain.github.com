00086 void dms_delete_vertex_internal(dms_vertex_t v)
00087 {
00088     dms_half_child_t c;
00089     while ((c = dms_half_children(v)))
00090         dms_delete_half_child(c);
00091     dms_half_edge_t h;
00092     dms_edge_t e;
00093     for (enum dms_direction d=0; d < 2; ++d)
00094     {
00095         while ((h = dms_half_edges(v,d)))
00096             dms_delete_half_edge(h);
00097         while ((e = dms_edges(v,d)))
00098             dms_delete_edge(e);
00099     }
00100     destroy_attributed_data(dms_vertex_data(v));
00101     remove_cut_vertex(v);
00102     reallocate(v,0);
