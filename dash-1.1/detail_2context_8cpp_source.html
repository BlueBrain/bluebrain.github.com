104 Context::~Context()
105 {
106  LBASSERTINFO( commit_.empty(), "Destroyed context has active changes");
107 
108  freeSlots_.push_back( slot_ );
109  if( slot_ == 0 ) // main context dtor
110  {
111  LBASSERTINFO( int32_t( freeSlots_.size( )) == numSlots_,
112  "Active context during main context destruction? : " <<
113  freeSlots_.size() << " != " << numSlots_ );
114  freeSlots_.clear();
115  numSlots_ = 0;
116  }
117 }
