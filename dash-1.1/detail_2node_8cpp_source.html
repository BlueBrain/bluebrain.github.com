60 Node::~Node()
61 {
62  LBASSERT( node_ == 0 );
63  while( !children_->empty( ))
64  {
65  // Don't use erase, generates Change
66  dash::NodePtr child = children_->back();
67  NodePtr childImpl = child->getImpl();
68  ParentsCtxPtr& parents = childImpl->parents_;
69  ParentsIter j = std::find( parents.getMutable().begin(),
70  parents.getMutable().end(), this );
71  LBASSERT( j != parents->end( ));
72 
73  parents.getMutable().erase( j );
74  children_.getMutable().pop_back();
75  }
76 
77  attributes_.getMutable().clear();
78 }

89 bool Node::operator == ( const Node& rhs ) const
90 {
91  if( this == &rhs )
92  return true;
93 
94  if( parents_->size() != rhs.parents_->size() ||
95  children_->size() != rhs.children_->size() ||
96  attributes_->size() != rhs.attributes_->size( ))
97  {
98  return false;
99  }
100 
101  Attributes::const_iterator it = attributes_->begin();
102  Attributes::const_iterator rhsIt = rhs.attributes_->begin();
103  for( ; it != attributes_->end() && rhsIt != rhs.attributes_->end();
104  ++it, ++rhsIt )
105  {
106  if( **it != **rhsIt )
107  return false;
108  }
109 
110  return true;
111 }

141 bool Node::erase( dash::NodePtr child )
142 {
143  ChildrenIter i = std::find( children_.getMutable().begin(),
144  children_.getMutable().end(), child);
145  if( i == children_.getMutable().end( ))
146  return false;
147 
148  ContextChange change( Change::NODE_ERASE, this, child );
149  Context::getCurrent().addChange( change );
150 
151  NodePtr childImpl = child->getImpl();
152 
153  ParentsCtxPtr& parents = childImpl->parents_;
154  ParentsIter j = std::find( parents.getMutable().begin(),
155  parents.getMutable().end(), this );
156  LBASSERT( j != parents.getMutable().end( ));
157 
158  parents.getMutable().erase( j );
159  children_.getMutable().erase( i );
160  return true;
