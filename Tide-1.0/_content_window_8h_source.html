272  template< class Archive >
273  void serialize_members_xml( Archive & ar, const unsigned int version )
274  {
275  ar & boost::serialization::make_nvp( "content", content_ );
276  if( version < 1 )
277  {
278  int contentWidth = 0, contentHeight = 0;
279  ar & boost::serialization::make_nvp( "contentWidth", contentWidth );
280  ar & boost::serialization::make_nvp( "contentHeight", contentHeight );
281  }
282  ar & boost::serialization::make_nvp( "coordinates", coordinates_ );
283  if( version < 3 )
284  {
285  QRectF backup;
286  ar & boost::serialization::make_nvp( "coordinatesBackup", backup );
287  }
288  QPointF zoomCenter = content_->getZoomRect().center();
289  qreal zoom = 1.0 / content_->getZoomRect().width();
290  ar & boost::serialization::make_nvp( "centerX", zoomCenter.rx( ));
291  ar & boost::serialization::make_nvp( "centerY", zoomCenter.ry( ));
292  ar & boost::serialization::make_nvp( "zoom", zoom );
293  if( version >= 3 )
294  ar & boost::serialization::make_nvp( "focused", focused_ );
295  QRectF zoomRect;
296  zoomRect.setSize( QSizeF( 1.0/zoom, 1.0/zoom ));
297  zoomRect.moveCenter( zoomCenter );
298  content_->setZoomRect( zoomRect );
299  if( version < 1 )
300  {
301  int controlState = 0;
302  ar & boost::serialization::make_nvp( "controlState", controlState );
303  }
304  ar & boost::serialization::make_nvp( "windowState", windowState_ );
305  }
