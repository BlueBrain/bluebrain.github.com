133 template< typename T >
134 inline std::ostream& operator << ( std::ostream& os,
135  const boost::multi_array< T, 2 >& data )
136 {
137  for( size_t i = 0; i < data.shape()[0]; ++i )
138  {
139  for( size_t j = 0; j < data.shape()[1]; ++j )
140  os << data[i][j] << " ";
141  os << std::endl;
142  }
143  return os;
144 }
