84 template <typename T, typename Container>
85 T get(const Container& container)
86 {
87  std::istringstream iss{std::istringstream::binary};
88 #ifdef __APPLE__
89  // pubsetbuf() does nothing on OSX (and probably on Windows too).
90  // so an intermediate copy of the data is required.
91  const std::string dataStr(container.data(), container.size());
92  iss.str(dataStr);
93 #else
94  iss.rdbuf()->pubsetbuf(const_cast<char*>(container.data()),
95  container.size());
96 #endif
97  T object;
98  {
99  boost::archive::binary_iarchive ia{iss};
100  ia >> object;
101  }
102  return object;
103 }

180 template <typename T>
181 bool fromXmlFile(T& object, const std::string& filename)
182 {
183  std::ifstream ifs(filename);
184  if (!ifs.good())
185  return false;
186  try
187  {
188  boost::archive::xml_iarchive ia(ifs);
189  ia >> BOOST_SERIALIZATION_NVP(object);
190  }
191  catch (const boost::archive::archive_exception& e)
192  {
193  print_log(LOG_ERROR, LOG_CONTENT,
194  "Could not restore from file: '%s': %s", filename.c_str(),
195  e.what());
196  return false;
197  }
198  catch (const std::exception& e)
199  {
200  print_log(LOG_ERROR, LOG_CONTENT,
201  "Could not restore from file '%s'',"
202  "wrong file format: %s",
203  filename.c_str(), e.what());
204  return false;
205  }
206  return true;
207 }
