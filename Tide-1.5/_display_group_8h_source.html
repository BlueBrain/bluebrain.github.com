212  template <class Archive>
213  void serialize_members_xml(Archive& ar, const unsigned int version)
214  {
215  // clang-format off
216  if (version == FIRST_DISPLAYGROUP_VERSION)
217  {
218  // legacy field, ignored but not removed yet to maintain full
219  // compatibility of sessions with previous Tide versions < 1.3.
220  bool _titles = true;
221  ar & boost::serialization::make_nvp("showWindowTitles", _titles);
222  }
223  ar & boost::serialization::make_nvp("contentWindows", _windows);
224  ar & boost::serialization::make_nvp("coordinates", _coordinates);
225  // clang-format on
226  }
227 
229  void serialize_for_xml(boost::archive::xml_iarchive& ar,
230  const unsigned int version)
231  {
232  serialize_members_xml(ar, version);
233  for (const auto& window : _windows)
234  {
235  if (window->isFocused())
236  _focusedWindows.insert(window);
237  if (window->isPanel())
238  _panels.insert(window);
239  // Make sure windows are not in an undefined state
240  window->setState(Window::NONE);
241  }
242  }
243 
245  void serialize_for_xml(boost::archive::xml_oarchive& ar,
246  const unsigned int version)
247  {
248  serialize_members_xml(ar, version);
249  }
