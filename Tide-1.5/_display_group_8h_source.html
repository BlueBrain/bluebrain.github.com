210  void serialize_members_xml(Archive& ar, const unsigned int version)
211  {
212  // clang-format off
213  if (version == FIRST_DISPLAYGROUP_VERSION)
214  {
215  // legacy field, ignored but not removed yet to maintain full
216  // compatibility of sessions with previous Tide versions < 1.3.
217  bool _titles = true;
218  ar & boost::serialization::make_nvp("showWindowTitles", _titles);
219  }
220  ar & boost::serialization::make_nvp("contentWindows", _contentWindows);
221  ar & boost::serialization::make_nvp("coordinates", _coordinates);
222  // clang-format on
223  }
224 
226  void serialize_for_xml(boost::archive::xml_iarchive& ar,
227  const unsigned int version)
228  {
229  serialize_members_xml(ar, version);
230  for (const auto& window : _contentWindows)
231  {
232  if (window->isFocused())
233  _focusedWindows.insert(window);
234  if (window->isPanel())
235  _panels.insert(window);
236  // Make sure windows are not in an undefined state
237  window->setState(ContentWindow::NONE);
238  }
239  }
240 
242  void serialize_for_xml(boost::archive::xml_oarchive& ar,
243  const unsigned int version)

245  serialize_members_xml(ar, version);
