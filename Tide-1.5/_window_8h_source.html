301  template <class Archive>
302  void serialize_members_xml(Archive& ar, const unsigned int version)
303  {
304  // clang-format off
305  if (version < 1)
306  {
307  int contentWidth = 0, contentHeight = 0;
308  ar & boost::serialization::make_nvp("contentWidth", contentWidth);
309  ar & boost::serialization::make_nvp("contentHeight", contentHeight);
310  }
311  ar & boost::serialization::make_nvp("coordinates", _coordinates);
312  if (version < 3)
313  {
314  QRectF backup;
315  ar & boost::serialization::make_nvp("coordinatesBackup", backup);
316  }
317  QPointF zoomCenter = _content->getZoomRect().center();
318  qreal zoom = 1.0 / _content->getZoomRect().width();
319  ar & boost::serialization::make_nvp("centerX", zoomCenter.rx());
320  ar & boost::serialization::make_nvp("centerY", zoomCenter.ry());
321  ar & boost::serialization::make_nvp("zoom", zoom);
322  if (version >= 3)
323  {
324  bool focused = isFocused();
325  ar & boost::serialization::make_nvp("focused", focused);
326  if (focused)
327  setMode(WindowMode::FOCUSED);
328  }
329  QRectF zoomRect;
330  zoomRect.setSize(QSizeF(1.0 / zoom, 1.0 / zoom));
331  zoomRect.moveCenter(zoomCenter);
332  _content->setZoomRect(zoomRect);
333  if (version < 1)
334  {
335  int controlState = 0;
336  ar & boost::serialization::make_nvp("controlState", controlState);
337  }
338  ar & boost::serialization::make_nvp("windowState", _state);
339  // clang-format on
340  }
