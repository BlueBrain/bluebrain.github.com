84  template <typename T>
85  inline void setProperty(const std::string &name, const T &t)
86  {
87  auto property = findProperty(name);
88  if (!property)
89  {
90  _properties.push_back(
91  std::make_shared<Property>(name, getType<T>()));
92  property = _properties[_properties.size() - 1].get();
93  }
94  property->set(t);
95  }

98  void setProperty(const Property &newProperty)

104  }

140  Property *findProperty(const std::string &name)

144  [&](const auto &p) { return p->name == name; });
