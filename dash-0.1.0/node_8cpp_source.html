00069 template< class N, class A > inline
00070 VisitorResult accept_( N* node, NodeVisitor< lunchbox::RefPtr< N >,
00071                                              lunchbox::RefPtr< A > >& visitor )
00072 {
00073     switch( visitor.visitDown( node ))
00074     {
00075       case ACCEPT_CONTINUE: break;
00076       case ACCEPT_PRUNE: return visitor.visitUp( node );
00077       case ACCEPT_TERMINATE: return ACCEPT_TERMINATE;
00078     }
00079 
00080     for( detail::ChildIterator< N > i = node->childrenBegin();
00081          i < node->childrenEnd(); ++i )
00082     {
00083         switch( i->accept( visitor ))
00084         {
00085           case ACCEPT_CONTINUE: break;
00086           case ACCEPT_PRUNE: return visitor.visitUp( node );
00087           case ACCEPT_TERMINATE: return ACCEPT_TERMINATE;
00088         }
00089     }
00090 
00091     for( detail::AttributeIterator< N, A > i = node->attributesBegin();
00092          i < node->attributesEnd(); ++i )
00093     {
00094         switch( visitor.visit( *i ))
00095         {
00096           case ACCEPT_CONTINUE: break;
00097           case ACCEPT_PRUNE:
00098               return visitor.visitUp( node );
00099           case ACCEPT_TERMINATE: return ACCEPT_TERMINATE;
00100         }
00101     }
00102     return visitor.visitUp( node );
00103 }
00104 }
