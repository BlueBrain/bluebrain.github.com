00076 template< class N, class A > inline
00077 VisitorResult accept_( N* node, NodeVisitor< lunchbox::RefPtr< N >,
00078                                              lunchbox::RefPtr< A > >& visitor )
00079 {
00080     switch( visitor.visitDown( node ))
00081     {
00082       case ACCEPT_CONTINUE: break;
00083       case ACCEPT_PRUNE: return visitor.visitUp( node );
00084       case ACCEPT_TERMINATE: return ACCEPT_TERMINATE;
00085     }
00086 
00087     for( detail::ChildIterator< N > i = node->childrenBegin();
00088          i < node->childrenEnd(); ++i )
00089     {
00090         switch( i->accept( visitor ))
00091         {
00092           case ACCEPT_CONTINUE: break;
00093           case ACCEPT_PRUNE: return visitor.visitUp( node );
00094           case ACCEPT_TERMINATE: return ACCEPT_TERMINATE;
00095         }
00096     }
00097 
00098     for( detail::AttributeIterator< N, A > i = node->attributesBegin();
00099          i < node->attributesEnd(); ++i )
00100     {
00101         switch( visitor.visit( *i ))
00102         {
00103           case ACCEPT_CONTINUE: break;
00104           case ACCEPT_PRUNE:
00105               return visitor.visitUp( node );
00106           case ACCEPT_TERMINATE: return ACCEPT_TERMINATE;
00107         }
00108     }
00109     return visitor.visitUp( node );
00110 }
00111 }
