00073 Context& Context::getMain( const int argc, char** argv )
00074 {
00075     lunchbox::ScopedFastWrite mutex( getInitLock_( ));
00076     if( !_mainContext )

00078         LBASSERT( lunchbox::Version::check( ));
00079         if( !lunchbox::Version::check()  || !lunchbox::init( argc, argv ))
00080             LBERROR << "Lunchbox initialization failed" << std::endl;

00082                << std::boolalpha << std::endl;

00087         LBWARN << "Main context already initialized, ignoring command line "
